LINKED LIST

Advantages

Provides linear data structure solutions.
Easy to insert, delete node without reorganize the array.
Allocate dynamic memory

Disadvantages

Do not allow random access to data. So sometimes, we must traverse all node to do what we want (delete tail, insert tail)

Code

//define
struct LLintNode {
int Data;
struct LLintNode* Next;
};
//counts number of node
int LengthLL(LLNode* head) {
int length = 0;
while (head != NULL) {
++length;
head = head ->Next;
}
return length;
}
//insert tail
void AddLast(LLNode** head, int data) {
LLNode** tmp = head;
LLNode* NewNode;
NewNode = (LLNode*) malloc(sizeof(LLNode));
NewNode->Data = data;
NewNode->Next = NULL;
if ((*tmp) == NULL) {
(*tmp) = NewNode;
} else {
while ((*tmp)->Next !=NULL) {
tmp = &((*tmp)->Next);
}
(*tmp)->Next = NewNode;
}
}

//remove tail
void RemoveLast(LLNode** head) {
LLNode** tmp = head;
if ((*tmp) !=NULL) {
while ((*tmp)->Next != NULL) {
tmp = &((*tmp)->Next);
}
}
(*tmp) = NULL;
}
